11    internal class DeepCloneAttribute : TypeAspect
 

12    {
13        public override void BuildAspect( IAspectBuilder<INamedType> builder )
14        {
15            var typedMethod = builder.Advice.IntroduceMethod(
16                builder.Target,
17                nameof(CloneImpl),
18                whenExists: OverrideStrategy.Override,
19                buildMethod: m =>
20                {
21                    m.Name = "Clone";
22                    m.ReturnType = builder.Target;
23                } );
24
25            builder.Advice.ImplementInterface(
26                builder.Target,
27                typeof(ICloneable),
28                whenExists: OverrideStrategy.Ignore );
29        }
30
31        [Template( IsVirtual = true )]
32        public virtual dynamic CloneImpl()
33        {
34            IExpression baseCall;
35
36            if (!meta.Target.Method.IsOverride)
37            {
38                ExpressionFactory.Capture( meta.Base.MemberwiseClone(), out baseCall );
39            }
40            else
41            {
42                ExpressionFactory.Capture( meta.Target.Method.Invoke(), out baseCall );
43            }
44
45            var clone = meta.Cast( meta.Target.Type.ToNonNullableType(), baseCall )!;
46
47            // Select clonable fields.
48            var clonableFields =
49                meta.Target.Type.FieldsAndProperties.Where(
50                    f => f.IsAutoPropertyOrField.GetValueOrDefault() &&
51                         ( f.Type.Is( typeof(ICloneable) ) ||
52                           ( f.Type is INamedType fieldNamedType && fieldNamedType.Enhancements().GetAspects<DeepCloneAttribute>().Any() ) ) );
53
54            foreach (var field in clonableFields)
55            {
56                field.With( (IExpression) clone ).Value = meta.Cast( field.Type, ( (ICloneable)field.Value! ).Clone() );
57            }
58
59            return clone;
60        }
61
62        [InterfaceMember( IsExplicit = true )]
63        private object Clone()
64        {
65            return meta.This.Clone();
66        }
67    }
